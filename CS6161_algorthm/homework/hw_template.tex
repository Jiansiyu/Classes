Here is a \LaTeX template for you to formally present the pseudocode of your algorithm \ref{alg:example},  

\begin{algorithm}[h] 
   \caption{Bubble Sort}
   \label{alg:example}
\begin{algorithmic}
   \STATE {\bfseries Input:} data $x_i$, size $m$
   \REPEAT
   \STATE Initialize noChange $=$ \TRUE.
   \FOR{$i=1$ {\bfseries to} $m-1$}
   \IF{$x_i > x_{i+1}$}
   \STATE Swap $x_i$ and $x_{i+1}$
   \STATE noChange $=$ \FALSE
   \ENDIF
   \ENDFOR
   \UNTIL{\textup{noChange} is \TRUE }
\end{algorithmic}
\end{algorithm}

\end{proof}
